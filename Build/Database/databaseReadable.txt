CREATE DATABASE pw_stuff collate utf8_general_ci;

USE pw_stuff

CREATE TABLE User (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
passwordListID INT UNSIGNED
);

CREATE TABLE PasswordList (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
userID INT UNSIGNED
);

CREATE TABLE Password (
id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
passwordListID INT UNSIGNED,
title VARCHAR(100),
username VARCHAR(100),
password VARCHAR(100)
);
